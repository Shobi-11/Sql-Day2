-- 1. courses
CREATE TABLE courses (
    _id INT PRIMARY KEY AUTO_INCREMENT,
    title VARCHAR(255) NOT NULL,
    description TEXT,
    instructorId INT,
    FOREIGN KEY (instructorId) REFERENCES instructors(_id)
);

-- 2. instructors
CREATE TABLE instructors (
    _id INT PRIMARY KEY AUTO_INCREMENT,
    firstName VARCHAR(255) NOT NULL,
    lastName VARCHAR(255) NOT NULL,
    email VARCHAR(255) NOT NULL UNIQUE,
     
);

-- 3. students
CREATE TABLE students (
    _id INT PRIMARY KEY AUTO_INCREMENT,
    firstName VARCHAR(255) NOT NULL,
    lastName VARCHAR(255) NOT NULL,
    email VARCHAR(255) NOT NULL UNIQUE,
);

-- 4. enrollments
CREATE TABLE enrollments (
    _id INT PRIMARY KEY AUTO_INCREMENT,
    studentId INT,
    courseId INT,
    enrollmentDate DATE NOT NULL,
    FOREIGN KEY (studentId) REFERENCES students(_id),
    FOREIGN KEY (courseId) REFERENCES courses(_id)
);

-- 5. users
CREATE TABLE users (
    _id INT PRIMARY KEY AUTO_INCREMENT,
    username VARCHAR(255) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,
    role VARCHAR(255) NOT NULL
);

-- 6. reviews
CREATE TABLE reviews (
    _id INT PRIMARY KEY AUTO_INCREMENT,
    courseId INT,
    studentId INT,
    rating INT NOT NULL,
    comment TEXT,
    FOREIGN KEY (courseId) REFERENCES courses(_id),
    FOREIGN KEY (studentId) REFERENCES students(_id)
);

-- 7. materials
CREATE TABLE materials (
    _id INT PRIMARY KEY AUTO_INCREMENT,
    courseId INT,
    title VARCHAR(255) NOT NULL,
    description TEXT,
    fileUrl VARCHAR(255),
    FOREIGN KEY (courseId) REFERENCES courses(_id)
);

-- 8. announcements
CREATE TABLE announcements (
    _id INT PRIMARY KEY AUTO_INCREMENT,
    courseId INT,
    title VARCHAR(255) NOT NULL,
    content TEXT NOT NULL,
    date DATE NOT NULL,
    FOREIGN KEY (courseId) REFERENCES courses(_id)
);

-- 9. activities
CREATE TABLE activities (
    _id INT PRIMARY KEY AUTO_INCREMENT,
    title VARCHAR(255) NOT NULL,
    description TEXT,
    date DATE NOT NULL,
    courseId INT,
    FOREIGN KEY (courseId) REFERENCES courses(_id)
);

-- 10. sessions_roadmap
CREATE TABLE sessions_roadmap (
    _id INT PRIMARY KEY AUTO_INCREMENT,
    sessionNumber INT NOT NULL,
    date DATE NOT NULL,
    courseId INT,
    FOREIGN KEY (courseId) REFERENCES courses(_id)
);

-- 11. topics_heading
CREATE TABLE topics_heading (
    _id INT PRIMARY KEY AUTO_INCREMENT,
    title VARCHAR(255) NOT NULL,
    courseId INT,
    FOREIGN KEY (courseId) REFERENCES courses(_id)
);

-- 12. contents
CREATE TABLE contents (
    _id INT PRIMARY KEY AUTO_INCREMENT,
    title VARCHAR(255) NOT NULL,
    description TEXT,
    courseId INT,
    topicId INT,
    FOREIGN KEY (courseId) REFERENCES courses(_id),
    FOREIGN KEY (topicId) REFERENCES topics_heading(_id)
);

-- 13. navbar_menu
CREATE TABLE navbar_menu (
    _id INT PRIMARY KEY AUTO_INCREMENT,
    label VARCHAR(255) NOT NULL,
    link VARCHAR(255) NOT NULL,
    orderNum INT NOT NULL,
    courseId INT,
    FOREIGN KEY (courseId) REFERENCES courses(_id)
);

-- 14. recording_links
CREATE TABLE recording_links (
    _id INT PRIMARY KEY AUTO_INCREMENT,
    title VARCHAR(255) NOT NULL,
    link VARCHAR(255) NOT NULL,
    date DATE NOT NULL,
    courseId INT,
    sessionId INT,
    FOREIGN KEY (courseId) REFERENCES courses(_id),
    FOREIGN KEY (sessionId) REFERENCES sessions_roadmap(_id)
);

-- 15. feedback
CREATE TABLE feedback (
    _id INT PRIMARY KEY AUTO_INCREMENT,
    courseId INT,
    studentId INT,
    rating INT NOT NULL,
    comment TEXT,
    date DATE NOT NULL,
    FOREIGN KEY (courseId) REFERENCES courses(_id),
    FOREIGN KEY (studentId) REFERENCES students(_id)
);

-- 16. additional_sessions
CREATE TABLE additional_sessions (
    _id INT PRIMARY KEY AUTO_INCREMENT,
    title VARCHAR(255) NOT NULL,
    description TEXT,
    date DATE NOT NULL,
    courseId INT,
    FOREIGN KEY (courseId) REFERENCES courses(_id)
);
